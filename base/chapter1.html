import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
plt.figure()
xs = list(range(11))
ys = [x**2 for x in xs]

ax = sns.lineplot(x=xs, y=ys)
ax.set_title("X^2")
ax.set_xlabel("x")
ax.set_ylabel("y")

filename = "chapter1/line.png"
plt.savefig(filename)
plt.close()

return filename
